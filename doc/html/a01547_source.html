00198         *TMR_ENBL     &= ~(TMR1);             /* tmrs reset to enabled */

00328 /*******************************************************************************
00329         KBI2 = GPIO24
00330         KBI3 = GPIO25
00331         KBI5 = GPIO27
00332         KBI2, KBI3 and KBI5 are connected to the CHRG output from the Li-Ion charger
00333                 KBI5 is the input that gives the state of charging
00334                 KBI2/3 are used to determine what that state is - KBI2 is a !OE line to a buffer with input given by KBI3
00335                         KBI2=1, KBI3=X =>
00336                                         KBI5==0 => charging (for all values of charging current)
00337                                         KBI5==1 => charging complete
00338                                 
00339                         KBI2=0, KBI3=1 =>
00340                                         KBI5==0 => charging (current > 10% of full scale)
00341                                         KBI5==1 => charging but current has dropped below 10% of full scale
00342 
00343   ADC0 = GPIO30
00344                 ADC0 is connected to an op-amp set up as a voltage follower (unity gain buffer)
00345                 It measures the voltage given by a voltage divider set to divide VBAT in the
00346     ratio 1.2 : 3.3 (i.e. to give VBAT * 3.3/4.5 as input to the ADC, remembering that
00347     full scale voltage on ADC is 3.3V)
00348 ***
00349         Initially, 
00350                 GPIO24 = o/p
00351                 GPIO25 = o/p
00352                 GPIO27 = i/p
00353                 GPIO30 = i/p - ADC
00354 *******************************************************************************/
00355 void charge_status_init(void) {
00356         GPIO->FUNC_SEL.CHARGE_STATUS_OE         = 0; // Default mode is GPIO
00357         GPIO->FUNC_SEL.CHARGE_STATUS_OUT        = 0;
00358         GPIO->FUNC_SEL.CHARGE_STATUS_IN         = 0;
00359         GPIO->FUNC_SEL.VBAT_IN                                          = 1; // It's an analogue input
00360 
00361         GPIO->PAD_DIR.CHARGE_STATUS_OE          = 1; // Set as output
00362         GPIO->PAD_DIR.CHARGE_STATUS_OUT         = 1;
00363         GPIO->PAD_DIR.CHARGE_STATUS_IN          = 0; // Set inputs
00364         GPIO->PAD_DIR.VBAT_IN                                                   = 0;
00365 
00366         CHARGE_STATUS_OE_HIGH;                                                           // Initially set the OE high, so value of CHARGE_STATUS_OUT is irrelevant
00367         CHARGE_STATUS_OUT_LOW;                                                           // But set it anyway
00368 
00369         // But leave VBAT ADC channel unconnected to any A2D converter for now
00370 }
