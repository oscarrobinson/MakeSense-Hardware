00228   ut was read from the device via I2C and fed into the right calc path for either SMD500 or BMP085
00229   \param ut parameter ut read from device
00230   \return temperature in steps of 0.1 deg celsius

00234 //short bmp085_calc_temperature(unsigned long ut) 
00235 short bmp085_get_temperature(unsigned long ut) 
00236 {
00237   short temperature;
00238   long x1,x2,x3,x4,y2,y3,y4;    
00239 
00240   if (p_bmp085->sensortype==BOSCH_PRESSURE_BMP085) {
00241     x1 = (((long) ut - (long) p_bmp085->cal_param.ac6) * (long) p_bmp085->cal_param.ac5) >> 15;
00242     x2 = ((long) p_bmp085->cal_param.mc << 11) / (x1 + p_bmp085->cal_param.md);
00243     p_bmp085->param_b5 = x1 + x2;
00244   }
00245   else // SMD500
00246   {
00247     if(p_bmp085->smd500_t_resolution==SMD500_T_RESOLUTION_16BIT)   // check for SMD500 temp resolution mode
00248       x1 = ((unsigned long) ((( ut * SMD500_PARAM_MJ) >> 16) - p_bmp085->cal_param.ac6));
00249     else //13BIT
00250       x1 = ut * 8  - p_bmp085->cal_param.ac6;
00251     
00252 
00253     x2 = (x1 * x1) >> 13;
00254     y2 = (SMD500_PARAM_MB * x2) >> 13;
00255     x3 = (x2 * x1) >> 16;
00256     y3 = (SMD500_PARAM_MC * x3) >> 12;
00257     x4 = (x2 * x2) >> 16;
00258     y4 = (SMD500_PARAM_MD * x4) >> 14;
00259 
00260     p_bmp085->param_b5 = (((p_bmp085->cal_param.ac5) * ( (2*x1) + y2 + y3 + y4))) >> 13; // temperature in (1/160)°C
00261   }
00262   temperature = ((p_bmp085->param_b5 + 8) >> 4);  // temperature in 0.1°C
00263 
00264   return (temperature);

00267 /** calculate pressure from up
00268   up was read from the device via I2C and fed into the right calc path for either SMD500 or BMP085
00269   In case of SMD500 value averaging is done in this function, in case of BMP085 averaging is done through oversampling by the sensor IC
00270 
00271   \param ut parameter ut read from device
00272   \return temperature in steps of 1.0 Pa
00273   \see bmp085_read_up()
00274 */

00276 //long bmp085_calc_pressure(unsigned long up)
00277 long bmp085_get_pressure(unsigned long up)
00278 {
00279    long pressure,x1,x2,x3,b3,b6;
00280    unsigned long b4, b7;
00281    
00282    b6 = p_bmp085->param_b5 - 4000;
00283    //*****calculate B3************
00284    x1 = (b6*b6) >> 12;           
00285    x1 *= p_bmp085->cal_param.b2;
00286    x1 >>=11;
00287 
00288    x2 = (p_bmp085->cal_param.ac2*b6);
00289    x2 >>=11;
00290 
00291    x3 = x1 +x2;
00292 
00293         b3 = (((((long)p_bmp085->cal_param.ac1 )*4 + x3) <<p_bmp085->oversampling_setting) + 2) >> 2;
00294 
00295    //*****calculate B4************
00296    x1 = (p_bmp085->cal_param.ac3* b6) >> 13;
00297    x2 = (p_bmp085->cal_param.b1 * ((b6*b6) >> 12) ) >> 16;
00298    x3 = ((x1 + x2) + 2) >> 2;
00299    b4 = (p_bmp085->cal_param.ac4 * (unsigned long) (x3 + 32768)) >> 15;
00300      
00301    b7 = ((unsigned long)(up - b3) * (50000>>p_bmp085->oversampling_setting));   
00302    if (b7 < 0x80000000)
00303    {
00304      pressure = (b7 << 1) / b4;
00305    }
00306    else
00307    { 
00308      pressure = (b7 / b4) << 1;
00309    }
00310    
00311    x1 = pressure >> 8;
00312    x1 *= x1;
00313    x1 = (x1 * SMD500_PARAM_MG) >> 16;
00314    x2 = (pressure * SMD500_PARAM_MH) >> 16;
00315    pressure += (x1 + x2 + SMD500_PARAM_MI) >> 4;        // pressure in Pa  
00316 
00317    return (pressure);
00318 }
