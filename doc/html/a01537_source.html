00150 /*******************************************************************************
00151 * Function Name  : LSM330DLC_SetMode_A
00152 * Description    : Sets LSM330DLC Operating Mode Acc
00153 * Input          : Modality (NORMAL, LOW_POWER, POWER_DOWN)
00154 * Output         : None
00155 * Return         : Status [MEMS_ERROR, MEMS_SUCCESS]
00156 *******************************************************************************/
00157 status_t LSM330DLC_SetMode_A(LSM330DLC_Mode_A_t md) {
00158   u8_t value;
00159   u8_t value2;
00160   static   u8_t ODR_old_value;
00161  
00162   if( !LSM330DLC_ReadReg(LSM330DLC_MEMS_I2C_ADDRESS_A,LSM330DLC_CTRL_REG1_A, &value) )
00163     return MEMS_ERROR;
00164   
00165   if( !LSM330DLC_ReadReg(LSM330DLC_MEMS_I2C_ADDRESS_A,LSM330DLC_CTRL_REG4_A, &value2) )
00166     return MEMS_ERROR;
00167   
00168   if((value & 0xF0)==0) value = value | (ODR_old_value & 0xF0); //if it comes from POWERDOWN  
00169     
00170   switch(md) {
00171   
00172   case LSM330DLC_POWER_DOWN_A:
00173     ODR_old_value = value;
00174     value &= 0x0F;
00175     break;
00176           
00177   case LSM330DLC_NORMAL_A:
00178     value &= 0xF7;
00179     value |= (MEMS_RESET<<LSM330DLC_LPEN_A);
00180     value2 &= 0xF7;
00181     value2 |= (MEMS_SET<<LSM330DLC_HR_A);   //set HighResolution_BIT
00182     break;
00183           
00184   case LSM330DLC_LOW_POWER_A:           
00185     value &= 0xF7;
00186     value |=  (MEMS_SET<<LSM330DLC_LPEN_A);
00187     value2 &= 0xF7;
00188     value2 |= (MEMS_RESET<<LSM330DLC_HR_A); //reset HighResolution_BIT
00189     break;
00190           
00191   default:
00192     return MEMS_ERROR;
00193   }
00194   
00195   if( !LSM330DLC_WriteReg(LSM330DLC_MEMS_I2C_ADDRESS_A,LSM330DLC_CTRL_REG1_A, value) )
00196     return MEMS_ERROR;
00197   
00198   if( !LSM330DLC_WriteReg(LSM330DLC_MEMS_I2C_ADDRESS_A,LSM330DLC_CTRL_REG4_A, value2) )
00199     return MEMS_ERROR;  
00200    
00201   return MEMS_SUCCESS;
00202 }
