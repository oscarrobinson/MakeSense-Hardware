00632 /* UBX parsing */
00633 void gps_ubx_parse( uint8_t c ) {
00634   if (gps_ubx.status < GOT_PAYLOAD) {
00635     gps_ubx.ck_a += c;
00636     gps_ubx.ck_b += gps_ubx.ck_a;
00637   }
00638   switch (gps_ubx.status) {
00639   case UNINIT:
00640     if (c == UBX_SYNC1) {
00641       gps_ubx.status++;
00642         }
00643     break;
00644   case GOT_SYNC1:
00645     if (c != UBX_SYNC2) {
00646       gps_ubx.error_last = GPS_UBX_ERR_OUT_OF_SYNC;
00647       goto error;
00648     }
00649     gps_ubx.ck_a = 0;
00650     gps_ubx.ck_b = 0;
00651     gps_ubx.status++;
00652     break;
00653   case GOT_SYNC2:
00654     if (gps_ubx.msg_available) {
00655       /* Previous message has not yet been parsed: discard this one */
00656       gps_ubx.error_last = GPS_UBX_ERR_OVERRUN;
00657       goto error;
00658     }
00659     gps_ubx.msg_class = c;
00660     gps_ubx.status++;
00661     break;
00662   case GOT_CLASS:
00663     gps_ubx.msg_id = c;
00664     gps_ubx.status++;
00665     break;
00666   case GOT_ID:
00667     gps_ubx.len = c;
00668     gps_ubx.status++;
00669     break;
00670   case GOT_LEN1:
00671     gps_ubx.len |= (c<<8);
00672     if (gps_ubx.len > GPS_UBX_MAX_PAYLOAD) {
00673       gps_ubx.error_last = GPS_UBX_ERR_MSG_TOO_LONG;
00674       goto error;
00675     }
00676     gps_ubx.msg_idx = 0;
00677     gps_ubx.status++;
00678     break;
00679   case GOT_LEN2:
00680     gps_ubx.msg_buf[gps_ubx.msg_idx] = c;

00684     }

00691     }
