00097 void lsm330dlc_decode(struct lsm330dlc_data *imu_data , uint8_t *buffer)
00098 {
00099         imu_data->acc_x =  ((buffer[1]<<8) + buffer[0]);
00100         imu_data->acc_y =  ((buffer[3]<<8) + buffer[2]);
00101         imu_data->acc_z =  ((buffer[5]<<8) + buffer[4]);

00103         imu_data->gyr_y =  ((buffer[9]<<8) + buffer[8]);
00104         imu_data->gyr_z = ((buffer[11]<<8) + buffer[10]);

00184

00190     break;
00191 
00192   default:
00193     return MEMS_ERROR;
00194   }
00195 
00196   LSM330DLC_WriteReg(LSM330DLC_I2C_ACC_ADDR,LSM330DLC_CTRL_REG1_A, value);
00197
