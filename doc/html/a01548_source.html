00595 /* UBX parsing */
00596 void gps_ubx_parse( uint8_t c ) {
00597   if (gps_ubx.status < GOT_PAYLOAD) {
00598     gps_ubx.ck_a += c;
00599     gps_ubx.ck_b += gps_ubx.ck_a;
00600   }
00601   switch (gps_ubx.status) {
00602   case UNINIT:
00603     if (c == UBX_SYNC1) {
00604       gps_ubx.status++;
00605         }
00606     break;
00607   case GOT_SYNC1:
00608     if (c != UBX_SYNC2) {
00609       gps_ubx.error_last = GPS_UBX_ERR_OUT_OF_SYNC;
00610       goto error;
00611     }
00612     gps_ubx.ck_a = 0;
00613     gps_ubx.ck_b = 0;
00614     gps_ubx.status++;
00615     break;
00616   case GOT_SYNC2:
00617     if (gps_ubx.msg_available) {
00618       /* Previous message has not yet been parsed: discard this one */
00619       gps_ubx.error_last = GPS_UBX_ERR_OVERRUN;
00620       goto error;
00621     }
00622     gps_ubx.msg_class = c;
00623     gps_ubx.status++;
00624     break;
00625   case GOT_CLASS:
00626     gps_ubx.msg_id = c;
00627     gps_ubx.status++;
00628     break;
00629   case GOT_ID:
00630     gps_ubx.len = c;
00631     gps_ubx.status++;
00632     break;
00633   case GOT_LEN1:
00634     gps_ubx.len |= (c<<8);
00635     if (gps_ubx.len > GPS_UBX_MAX_PAYLOAD) {
00636       gps_ubx.error_last = GPS_UBX_ERR_MSG_TOO_LONG;
00637       goto error;
00638     }
00639     gps_ubx.msg_idx = 0;
00640     gps_ubx.status++;
00641     break;
00642   case GOT_LEN2:
00643     gps_ubx.msg_buf[gps_ubx.msg_idx] = c;
00644     gps_ubx.msg_idx++;
00645     if (gps_ubx.msg_idx >= gps_ubx.len) {
00646       gps_ubx.status++;
00647     }
00648     break;
00649   case GOT_PAYLOAD:
00650     if (c != gps_ubx.ck_a) {
00651       gps_ubx.error_last = GPS_UBX_ERR_CHECKSUM;
00652 PRINTF("DEBUG: Checksum-a failed\n");
00653       goto error;
00654     }
00655     gps_ubx.status++;
00656     break;
00657   case GOT_CHECKSUM1:
00658     if (c != gps_ubx.ck_b) {
00659 PRINTF("DEBUG: Checksum-b failed\n");
00660       gps_ubx.error_last = GPS_UBX_ERR_CHECKSUM;
00661       goto error;
00662     }
00663 PRINTF("DEBUG: MSG received\n");
00664     gps_ubx.msg_available = TRUE;
00665     goto restart;
00666     break;
00667   default:
00668     gps_ubx.error_last = GPS_UBX_ERR_UNEXPECTED;
00669 PRINTF("DEBUG: UBX ERR UNEXPECTED\n");
00670     goto error;
00671   }
00672   return;
00673  error:
00674   gps_ubx.error_cnt++;
00675  restart:
00676   gps_ubx.status = UNINIT;
00677 
00678   // Added by Jagun
00679 //  gps_ubx.ck_a = 0;
00680 //  gps_ubx.ck_b = 0;
00681   return;
00682 }
