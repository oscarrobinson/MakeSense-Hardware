00052   printf("Setting fsync dlpf config...\n");
00053   uint8_t fsync_dlpf_config = 0b00000011; //FSYNC disabled and dlpf_cfg set to 3.
00054   cmd[0] = MPU9150_CONFIG;
00055   cmd[1] = fsync_dlpf_config;
00056   i2c_transmitinit( MPU9150_I2C_ADDR, 2, cmd );
00057   while(!i2c_transferred()) /* Wait for transfer */ ;

00060    * With current dlpf_cfg (3) the sample rate of the accelerometer is 1kHz and the 
00061    * gyroscope is 1kHz too. The sample rate divider has been set to 20 (register 20).
00062    * Sample rate = Gyroscope output rate / (1 + sample rate divider)
00063    *             =                  1000 / (1 + 19)
00064    *             =                       50Hz
00065    */
00066   /* Set up the fifo_config parameter and enable sensors if they are defined in the header file: */
00067   uint8_t fifo_config = 0b00000000;
00068   /*=== Trigger the gyroscopes self-test and set the scale range: (REG:27) ===*/
00069   if (INV_XYZ_GYRO) {
00070     fifo_config = fifo_config | 0b01110000;
00071     printf("Starting up gyroscope...\n");
00072     //Self test xyz and set full scale range to +-2000s:
00073     uint8_t gyro_config = 0b11111000;
00074     cmd[0] = MPU9150_GYRO_CONFIG;
00075     cmd[1] = gyro_config;
00076     i2c_transmitinit( MPU9150_I2C_ADDR, 2, cmd ) ;
00077     while(!i2c_transferred()) /* Wait for transfer */ ;
00078     printf("Gyroscope started!\n");
00079   }

00083      fifo_config = fifo_config | 0b00001000;
00084      printf("Starting up accelorometer...\n");
00085      uint8_t accel_config = 0b11111000; //self test xyz, with full scale range +-16g and reset hpf
00086      cmd[0] = MPU9150_ACCEL_CONFIG;
00087      cmd[1] = accel_config;

00089      while(!i2c_transferred()) /* Wait for transfer */ ;
00090      printf("Accelorometer started!\n");
00091   }
00092   
00093   /*=== Enable Temp on FIFO: ===*/
00094   if (INV_TEMP) {
00095     fifo_config = fifo_config | 0b10000000;
00096     printf("Adding thermometer to FIFO...\n");
00097   }
00098   
00099   if (INV_XYZ_ACCEL) {
00100         packet_size += 6;}
00101     if (INV_TEMP) {
00102         packet_size += 2;}
00103     if (INV_XYZ_GYRO) {
00104         packet_size += 6;}
00105     
00106     printf("packet size: %i\n", packet_size);
00107     //exit(0);
00108   
00109   /*
00110    * SKIPPING FREE FALL, MOTION DETECTION, ETC. CONFIGURATION
00111    */
00112   
00113    /*=== Setup FIFO queue: (REG:35) ===
00114     Bit order [1: Temp, 2-4: Gyro{x,y.z}, 5: Accel, 6-8: SLV{2,1,0}] */
00115   int BUF_SIZE = 9;
00116   char buffer[BUF_SIZE];
00117   buffer[BUF_SIZE - 1] = '\0';
00118   printf("Configuring FIFO with parameter %s...\n", int2bin(fifo_config, buffer, BUF_SIZE-1));
00119   cmd[0] = MPU9150_FIFO_EN;
00120   cmd[1] = fifo_config;
00121   i2c_transmitinit( MPU9150_I2C_ADDR, 2, cmd );
00122   while(!i2c_transferred()) /* Wait for transfer */ ;
00123   printf("FIFO configured!\n");
00124   
00125   /* Setup i2c master control (REG:36)
00126      First bit controls multi-master capability (not needed for us I think)
00127      Second bit controls wait for external sensors (to keep data in sync), which is not needed by us, because we do ot use external sensors
00128      Third bit controls FIFO_EN for salve 3 (other slaves controled in REG35)
00129      Fourth bit controls slave read and write mode (don't really know waht the implications of the different write modes are, think it's not significant for us?)
00130      Last four bits control the i2c master clock. We set the 8mhz clock divider to 19, so set this to 12 1100??? */
00131   printf("Setting up I2C master controller...\n");
00132   uint8_t i2cmc_config= 0b00001101;
00133   cmd[0] = MPU9150_I2C_MST_CTRL;
00134   cmd[1] = i2cmc_config;
00135   i2c_transmitinit( MPU9150_I2C_ADDR, 2, cmd );
00136   while(!i2c_transferred()) /* Wait for transfer */ ;
00137   printf("I2C master configured!\n");
00138   
00139   /* Setup user control */
00140   printf("Setting up user control...\n");
00141   uint8_t user_config= 0b01000100;
00142   cmd[0] = MPU9150_USER_CTRL;
00143   cmd[1] = user_config;
00144   i2c_transmitinit( MPU9150_I2C_ADDR, 2, cmd );
00145   while(!i2c_transferred()) /* Wait for transfer */ ;
00146   printf("User control configured!\n");
00147   
00148   /*
00149    * SLAVE CONTROL REGISTERS (37-53) WE DON'T NEED TO CONFIGURE (I THINK). 54 IS THE I2C MASTER CONTROLLER,
00150    * ALSO NOT USED BY US, 55 IS USED TO CONFIGURE INTERRUPTS (MAYBE WE SHOULD CONFIGURE), 56 ENABLES
00151    * INTERRUPTS, MAYBE WE SHOULD SET THE FIFO INTERRUPT TO ONE, THEN WAIT FOR THE INTERRUPT TO OCCOUR
00152    * AND THEN READ ALL SENSORS IN ONE GO AND WAIT FOR NEXT INTERRUPT. REG 58 CONTAINS THE INTERRUPT STATA.
00153    * 59-96 CONTAINS THE LATEST VALUES FROM THE SENOSRS, BUT WE READ THEM FROM THE FIFO INSTEAD. REG 97 IS
00154    * MOTION DETETION STATUS, 98-102 HAS THE DATAOUT FOR THE IC2 SLAVES, 103 CONTROLS IC2 MASTER DEALY, 104
00155    * CAN BE USED TO RESET THE PATHS FOR THE INTERNAL SENSORS, 105 IS MOTION DETECTION CONTROL, 106 USER CONTROL,
00156    * BUT WE ENABLE THE FIFO IN A DIFFERENT REGISTER, SO SON'T NEED TO REENABLE IT HERE. 107 AND 108 ARE
00157    * POWER MANAGEMENT.
00158    * NOW TO GET DATA, WE SHOULD LOOK AT REGISTERS 114 AND 115 TO GET THE COUNT OF NEW MEASUREMENTS AND THEN
00159    * READ THEM FROM THE FIFO QUEUE:
00160    */
