00033 void RTC_init()
00034 {       uint8_t buf;
00035 
00036   // At the point we power up, the initial values for the registers are:
00037   // ST       = 0                       -- we're not stopped, but OF is set because we just powered up
00038   // OF       = 1                       -- attempt to reset this, until 4 seconds have elapsed, then cycle ST
00039   // OFIE     = 0                       -- Oscillator fail interrupt enable.
00040   // OUT      = 1                       -- This will be the value on the IRQ/OUT pin cos OFIE, AFE and WD are all zero
00041   // AFE      = 0                       -- Alarm enable
00042   // SQWE     = 1                       -- Square wave enable
00043   // RS[3:0]  = 0x01    -- Frequency of square wave = 32.768 kHz
00044   // WATCHDOG = 0                       -- Watchdog off. Leave it off, because we have our own if wanted.
00045 
00046         // To set these registers, we need to make the NON-time related changes before
00047   // setting the time. Any write to the registers containing time info stops the clock
00048   //
00049 
00050         // Set stop flag ST to one and back as per datasheet advice to kick the ocscillator
00051   // First, read the register
00052         RTC_stop();
00053   RTC_start();
00054 
00055   // Turn the square wave off for now.
00056   //
00057   // First, read the register continaing the SQWE bit into buf
00058         M41T62_readReg(M41T62_REG_SQWE, &buf);
00059   //
00060   // Now clear the bit and send the rest back
00061         buf &= ~M41T62_SQWE_MASK;
00062         M41T62_writeReg(M41T62_REG_SQWE, buf);
00063 
00064   // Clear the oscillator fail (OF) flag. As per the datasheet,
00065         // we try this repeatedly until 4 seconds have elapsed and
00066   // then we blip the ST bit to one and zero in short succession
00067         // and try again
00068   //
00069   int done = 0;
00070   while (!done) {
00071         int now  = clock_time();
00072 
00073                 while ((clock_time() - now) < 4 * CLOCK_CONF_SECOND) {
00074                         // Set the OF bit to zero
00075                   // First, read the flags register
00076                         M41T62_readReg(M41T62_REG_FLAGS, &buf);
00077       //
00078                         // If the bit is zero, we're done
00079                         if ((buf & M41T62_OF_MASK) > 0) {
00080                                 done = 1;
00081                                 break;
00082                         }
00083                         //
00084                         // Now set it to zero
00085                         buf &= ~M41T62_OF_MASK;
00086                         M41T62_writeReg(M41T62_REG_FLAGS, buf);                 
00087                 }
00088 
00089                 if (done)
00090       break;
00091 
00092                 RTC_stop();
00093                 RTC_start();
00094         }
00095 
00096   // Clear all the bits in the registers that should be zero
00097   //
00098   RTC_resetRegisters();
00099   print_regs("Initial");
00100 
00101   enable_irq_kbi(5);                    // RTC IRQ
00102 }

00119 void RTC_setTime(struct RTC_time *t)
00120 { static uint8_t rxbuf[] = {0, 0, 0, 0, 0, 0, 0, 0};
00121   static uint8_t txbuf[] = {0, 0, 0, 0, 0, 0, 0, 0, 0};
00122 
00123   // Read the registers
00124         M41T62_readRegN(M41T62_REG_HUNDREDTHS, rxbuf, 8);
00125 
00126   // Clear out the bits we want to change
00127   rxbuf[0] &= 0;
00128   rxbuf[1] &= ~(M41T62_10SECONDS_MASK | M41T62_SECONDS_MASK);
00129   rxbuf[2] &= ~(M41T62_10MINUTES_MASK | M41T62_MINUTES_MASK);
00130   rxbuf[3] &= ~(M41T62_10HOURS_MASK   | M41T62_HOURS_MASK);
00131   rxbuf[5] &= ~(M41T62_10DAY_MASK     | M41T62_DAY_MASK);
00132   rxbuf[6] &= ~(M41T62_10MONTH_MASK   | M41T62_MONTH_MASK);
00133   rxbuf[7] &= 0;
00134 
00135   // Set the bits as we want them. Note that we
00136   // must construct a transmit buffer that contains
00137   // the register number as the first element, followed
00138   // by the data.
00139   //
00140   txbuf[0] = M41T62_REG_HUNDREDTHS;
00141   txbuf[1] = 0;         // Always reset to zero by chip anyway
00142   txbuf[2] = rxbuf[1] | M41T62_SET_SECONDS(t->seconds);
00143   txbuf[3] = rxbuf[2] | M41T62_SET_MINUTES(t->minutes);
00144   txbuf[4] = rxbuf[3] | M41T62_SET_HOURS(t->hours);
00145   txbuf[5] = rxbuf[4];
00146   txbuf[6] = rxbuf[5] | M41T62_SET_DAY(t->day);
00147   txbuf[7] = rxbuf[6] | M41T62_SET_MONTH(t->month);
00148   txbuf[8] = rxbuf[7] | M41T62_SET_YEAR(t->year);
00149 
00150   // And write them back
00151   //
00152   M41T62_writeRegN(txbuf, 9);
00153   print_regs("Set time");
00154 }

00372 { uint8_t buf;
00373 
00374   PROCESS_BEGIN();
00375  
00376   while(1) {
00377     PROCESS_YIELD();

00379     if (ev != PROCESS_EVENT_POLL) {
00380       continue;
