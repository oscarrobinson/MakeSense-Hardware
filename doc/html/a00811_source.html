00164 /*******************************************************************************
00165         KBI2 = GPIO24
00166         KBI3 = GPIO25
00167         KBI5 = GPIO27
00168         KBI2, KBI3 and KBI5 are connected to the CHRG output from the Li-Ion charger
00169                 KBI5 is the input that gives the state of charging
00170                 KBI2/3 are used to determine what that state is - KBI2 is a !OE line to a buffer with input given by KBI3
00171                         KBI2=1, KBI3=X =>
00172                                         KBI5==0 => charging (for all values of charging current)
00173                                         KBI5==1 => charging complete
00174                                 
00175                         KBI2=0, KBI3=1 =>
00176                                         KBI5==0 => charging (current > 10% of full scale)
00177                                         KBI5==1 => charging but current has dropped below 10% of full scale
00178 
00179   ADC0 = GPIO30
00180                 ADC0 is connected to an op-amp set up as a voltage follower (unity gain buffer)
00181                 It measures the voltage given by a voltage divider set to divide VBAT in the
00182     ratio 1.2 : 3.3 (i.e. to give VBAT * 3.3/4.5 as input to the ADC, remembering that
00183     full scale voltage on ADC is 3.3V)
00184 ***
00185         Initially, 
00186                 GPIO24 = o/p
00187                 GPIO25 = i/p
00188                 GPIO27 = i/p
00189                 GPIO30 = i/p - ADC
00190 *******************************************************************************/

00192 #define CHARGE_STATUS_OUT                                       GPIO_25
00193 #define CHARGE_STATUS_IN                                        GPIO_27
00194 #define CHARGE_STATUS_OE_LOW                    (GPIO->DATA_RESET.CHARGE_STATUS_OE = 1)
00195 #define CHARGE_STATUS_OE_HIGH                   (GPIO->DATA_SET.CHARGE_STATUS_OE = 1)
00196 #define CHARGE_STATUS_OUT_LOW                   (GPIO->DATA_RESET.CHARGE_STATUS_OUT = 1)
